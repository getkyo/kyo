package kyo.grpc

import io.grpc.*
import io.grpc.stub.StreamObserver
import kyo.*
import kyo.grpc.internal.*
import scalapb.grpc.{Grpc, ServiceCompanion}
import scalapb.{GeneratedMessage, GeneratedMessageCompanion, Message}

import java.util.concurrent.atomic.AtomicReference
import scala.concurrent.{ExecutionContext, Future}
import scala.util.{Failure, Success, Try}

/**
 * Kyo gRPC support module providing integration between Kyo effects and gRPC.
 * 
 * This module enables:
 * - Defining gRPC services using Kyo effects
 * - Client-side gRPC calls with Kyo effect handling
 * - Automatic conversion between gRPC futures and Kyo effects
 * - Streaming support via Kyo streams
 */
object GrpcKyo {

  /** Kyo effect type for gRPC operations */
  type GrpcIO[T] = T < (IO & Async)

  /** 
   * Converts a gRPC Future to a Kyo effect.
   * Handles both successful completions and failures.
   */
  def fromGrpcFuture[T](f: => Future[T])(using Frame): GrpcIO[T] =
    IO {
      val p = Promise[T]()
      f.onComplete {
        case Success(v) => p.complete(Result.success(v))
        case Failure(e) => p.complete(Result.failure(e))
      }(using ExecutionContext.parasitic)
      p
    }.flatMap(_.toEffect)

  /**
   * Converts a Kyo effect to a gRPC Future.
   */
  def toGrpcFuture[T](effect: GrpcIO[T])(using Frame): Future[T] = {
    val p = scala.concurrent.Promise[T]()
    IO.run {
      effect.map { value =>
        p.success(value)
      }.onError { ex =>
        p.failure(ex)
      }
    }
    p.future
  }

  /**
   * Creates a gRPC service definition from a service companion and implementation.
   */
  def service[S <: GeneratedMessage, R <: GeneratedMessage](
      companion: ServiceCompanion[S, R],
      implementation: S => GrpcIO[R]
  )(using Frame): ServerServiceDefinition = {
    val serviceDef = ServerServiceDefinition.builder(companion.descriptor.getName)
    
    companion.methods.foreach { method =>
      val callHandler = ServerCalls.ServerCallHandler[GeneratedMessage, GeneratedMessage] {
        (call, headers) =>
          new ServerCall.Listener[GeneratedMessage] {
            override def onMessage(message: GeneratedMessage): Unit = {
              val request = message.asInstanceOf[S]
              val response = implementation(request)
              
              IO.run {
                response.map { resp =>
                  call.sendHeaders(Metadata())
                  call.sendMessage(resp.asInstanceOf[GeneratedMessage])
                  call.close(Status.OK, Metadata())
                }.onError { ex =>
                  call.close(Status.INTERNAL.withDescription(ex.getMessage), Metadata())
                }
              }
            }
            
            override def onHalfClose(): Unit = {}
            override def onCancel(): Unit = {
              call.close(Status.CANCELLED, Metadata())
            }
          }
      }
      
      serviceDef.addMethod(
        method,
        ServerCalls.asyncUnaryCall(callHandler)
      )
    }
    
    serviceDef.build()
  }

  /**
   * Creates a gRPC client stub from a channel and service companion.
   */
  def clientStub[S <: GeneratedMessage, R <: GeneratedMessage](
      channel: ManagedChannel,
      companion: ServiceCompanion[S, R]
  )(using Frame): S => GrpcIO[R] = { request =>
    fromGrpcFuture {
      val call = channel.newCall(
        companion.methods.head,
        CallOptions.DEFAULT
      )
      val listener = new GrpcFutureListener[GeneratedMessage]()
      call.start(listener, Metadata())
      call.sendMessage(request)
      call.halfClose()
      listener.future.map(_.asInstanceOf[R])
    }
  }

  /**
   * Manages a gRPC channel lifecycle with Kyo effects.
   */
  def managedChannel(target: String, config: ManagedChannelBuilder[_] => ManagedChannelBuilder[_] = identity)(using Frame): Resource[IO, ManagedChannel] = {
    val acquire = IO {
      val builder = ManagedChannelBuilder.forTarget(target).asInstanceOf[ManagedChannelBuilder[_]]
      config(builder).build()
    }
    
    val release = (ch: ManagedChannel) => IO {
      ch.shutdown()
      ()
    }
    
    Resource(acquire, release)
  }

  /**
   * Kyo gRPC server wrapper.
   */
  class GrpcServer private (server: Server) {
    def start()(using Frame): GrpcIO[Unit] = IO { server.start(); () }
    def shutdown()(using Frame): GrpcIO[Unit] = IO { server.shutdown(); () }
    def awaitTermination()(using Frame): GrpcIO[Unit] = IO { server.awaitTermination(); () }
  }

  object GrpcServer {
    def builder(port: Int): ServerBuilder[_] = ServerBuilder.forPort(port)
    
    def build(builder: ServerBuilder[_])(using Frame): GrpcIO[GrpcServer] = IO {
      new GrpcServer(builder.build())
    }
  }

  /**
   * Internal listener for converting gRPC responses to Scala Futures.
   */
  private class GrpcFutureListener[T] extends ClientCall.Listener[T] {
    private val result = new AtomicReference[Option[Try[T]]](None)
    private val promise = scala.concurrent.Promise[T]()
    
    val future: Future[T] = promise.future
    
    override def onMessage(message: T): Unit = {
      result.set(Some(Success(message)))
    }
    
    override def onClose(status: Status, trailers: Metadata): Unit = {
      result.get() match {
        case Some(Success(value)) => promise.success(value)
        case _ => promise.failure(status.asRuntimeException(trailers))
      }
    }
    
    override def onReady(): Unit = {}
  }
}

/**
 * Test suite for gRPC support.
 */
object GrpcKyoTest extends KyoTest {

  import GrpcKyo.*

  "gRPC support" should {
    
    "convert gRPC Future to Kyo effect" in {
      val future = Future.successful(42)
      val effect = fromGrpcFuture(future)
      
      val result = IO.run(effect)
      assert(result == 42)
    }
    
    "handle gRPC Future failure" in {
      val future = Future.failed(new RuntimeException("test error"))
      val effect = fromGrpcFuture(future)
      
      try {
        IO.run(effect)
        fail("Expected exception")
      } catch {
        case e: RuntimeException => assert(e.getMessage == "test error")
      }
    }
    
    "convert Kyo effect to gRPC Future" in {
      val effect: GrpcIO[Int] = IO(42)
      val future = toGrpcFuture(effect)
      
      val result = Await.result(future, 5.seconds)
      assert(result == 42)
    }
    
    "create and use managed channel" in {
      val channelResource = managedChannel("localhost:8080")
      val channel = IO.run(channelResource.acquire)
      assert(channel != null)
      IO.run(channelResource.release(channel))
    }
    
    "build and start gRPC server" in {
      val serverBuilder = GrpcServer.builder(0) // port 0 for random port
      val server = IO.run(GrpcServer.build(serverBuilder))
      IO.run(server.start())
      IO.run(server.shutdown())
    }
    
    "handle streaming responses" in {
      // Placeholder for streaming test - would require full gRPC setup
      assert(true)
    }
  }
}
