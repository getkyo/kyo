/**
 * Kyo gRPC support module providing integration with ScalaPB generated code.
 * 
 * This module enables efficient gRPC client and server implementations using
 * Kyo's effect system for resource management and concurrency.
 */
object Grpc {

  /**
   * Creates a gRPC client stub for the given service.
   * 
   * @param channel The gRPC channel to use
   * @param options Call options for the stub
   * @return A client stub wrapped in Kyo effects
   */
  def client[T <: AbstractStub[T]](
      channel: Channel,
      options: CallOptions = CallOptions.DEFAULT
  ): T = 
    throw new UnsupportedOperationException("Use generated client stubs instead")

  /**
   * Creates a gRPC server that handles requests using Kyo effects.
   * 
   * @param service The service definition
   * @param port The port to bind to
   * @return A Kyo effect that starts the server
   */
  def server(
      service: ServerServiceDefinition,
      port: Int = 8080
  ): Server = 
    ServerBuilder.forPort(port)
      .addService(service)
      .build()
      .start()

  /**
   * Kyo-based gRPC client stub that uses fibers for non-blocking I/O.
   */
  abstract class KyoClientStub[S <: KyoClientStub[S]](
      channel: Channel,
      options: CallOptions
  ) extends AbstractStub[S](channel, options) {

    /**
     * Performs a unary gRPC call.
     * 
     * @param method The method descriptor
     * @param request The request message
     * @param marshaller The response marshaller
     * @return A Kyo effect that yields the response
     */
    def unaryCall[Req, Res](
        method: MethodDescriptor[Req, Res],
        request: Req,
        marshaller: MessageMarshaller[Res]
    )(using Frame): Res < (IO & Async) = {
      IO {
        val call = newCall(method, options)
        Async.run {
          val listener = new KyoClientCallListener[Res]()
          call.start(listener, new KyoClientCallListener.KyoClientResponseListener(request, marshaller))
          listener.result
        }
      }.flatMap(identity)
    }

    /**
     * Performs a server-streaming gRPC call.
     * 
     * @param method The method descriptor
     * @param request The request message
     * @param marshaller The response marshaller
     * @return A Kyo stream of responses
     */
    def serverStreamingCall[Req, Res](
        method: MethodDescriptor[Req, Res],
        request: Req,
        marshaller: MessageMarshaller[Res]
    )(using Frame): Stream[Res, IO & Async] = {
      Stream(emit => {
        IO {
          val call = newCall(method, options)
          val listener = new KyoServerStreamingListener[Res](emit, marshaller)
          call.start(listener, new KyoClientCallListener.KyoClientResponseListener(request, marshaller))
          listener.done
        }.flatMap(identity)
      })
    }

    /**
     * Performs a client-streaming gRPC call.
     * 
     * @param method The method descriptor
     * @param requests A stream of requests
     * @param marshaller The response marshaller
     * @return A Kyo effect that yields the response
     */
    def clientStreamingCall[Req, Res](
        method: MethodDescriptor[Req, Res],
        requests: Stream[Req, Any],
        marshaller: MessageMarshaller[Res]
    )(using Frame): Res < (IO & Async) = {
      IO {
        val call = newCall(method, options)
        val listener = new KyoClientCallListener[Res]()
        call.start(listener, new KyoClientStreamingRequestListener(requests, call, marshaller))
        listener.result
      }.flatMap(identity)
    }

    /**
     * Performs a bidirectional streaming gRPC call.
     * 
     * @param method The method descriptor
     * @param requests A stream of requests
     * @param marshaller The response marshaller
     * @return A stream of responses
     */
    def bidiStreamingCall[Req, Res](
        method: MethodDescriptor[Req, Res],
        requests: Stream[Req, Any],
        marshaller: MessageMarshaller[Res]
    )(using Frame): Stream[Res, IO & Async] = {
      Stream(emit => {
        IO {
          val call = newCall(method, options)
          val listener = new KyoServerStreamingListener[Res](emit, marshaller)
          call.start(listener, new KyoClientStreamingRequestListener(requests, call, marshaller))
          listener.done
        }.flatMap(identity)
      })
    }

    override def build(channel: Channel, options: CallOptions): S
  }

  /**
   * Kyo-based gRPC server service definition.
   */
  abstract class KyoServerServiceDefinition {
    
    /**
     * Binds this service to a server.
     * 
     * @return The server service definition
     */
    def bindService(): ServerServiceDefinition
  }

  /**
   * Helper to create a unary server method handler.
   * 
   * @param handler The handler function
   * @return A server call handler
   */
  def unaryServerHandler[Req, Res](
      handler: Req => Res < (IO & Async)
  )(using 
      marshaller: MessageMarshaller[Res],
      descriptor: MethodDescriptor[Req, Res]
  ): ServerCallHandler[Req, Res] = {
    new ServerCallHandler[Req, Res] {
      def startCall(
          call: ServerCall[Req, Res],
          headers: Metadata
      ): Unit = {
        val listener = new KyoServerCallListener[Req, Res](call, handler, marshaller)
        call.request(1)
        call.setMessageCompression(false)
        call.sendHeaders(new Metadata())
        listener.start()
      }
    }
  }

  /**
   * Helper to create a server-streaming server method handler.
   * 
   * @param handler The handler function that returns a stream
   * @return A server call handler
   */

      handler: Req => Stream[Res, IO & Async]
  )(using 
      marshaller: MessageMarshaller[Res],
      descriptor: MethodDescriptor[Req, Res]

          call: ServerCall[Req, Res],

      }
    }
  }

  def clientStreamingServerHandler[Req, Res](

        call.request(1)
        call.setMessageCompression(false)
        call.sendHeaders(new Metadata())
        listener.start()
      }
    }
