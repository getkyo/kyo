/**
 * Kyo gRPC support module providing integration with ScalaPB generated code.
 * 
 * This module enables efficient gRPC client and server implementations using Kyo effects,
 * with automatic resource management and cancellation support.
 */
object Grpc:

  /**
   * Configuration for gRPC channels and servers.
   */
  case class Config(
      deadline: Option[Duration] = None,
      maxRetries: Int = 0,
      compression: Option[String] = None,
      interceptors: List[ClientInterceptor] = Nil
  )

  object Config:
    val default: Config = Config()

  /**
   * Creates a gRPC client stub for the given service.
   */
  def client[T <: AbstractStub[T]](
      channel: ManagedChannel,
      config: Config = Config.default
  )(using
      tag: ClassTag[T],
      stubFactory: GrpcStubFactory[T]
  ): T =
    val baseStub = stubFactory.create(channel)
    val withConfig = config.interceptors.foldLeft(baseStub) { (stub, interceptor) =>
      stub.asInstanceOf[AbstractStub[T]].withInterceptors(interceptor).asInstanceOf[T]
    }
    config.deadline match
      case Some(d) => withConfig.withDeadlineAfter(d.toNanos, java.util.concurrent.TimeUnit.NANOSECONDS)
      case None    => withConfig

  /**
   * Manages a gRPC channel lifecycle with Kyo resources.
   */
  def managedChannel(target: String, config: Config = Config.default): ManagedChannel < (IO & Resources) =
    Resources.ensure {
      IO {
        val builder = ManagedChannelBuilder.forTarget(target).usePlaintext()
        config.compression.foreach(builder.compression)
        val channel = builder.build()
        (channel, IO(channel.shutdown()))
      }
    }

  /**
   * Creates a gRPC server with Kyo resource management.
   */
  def server(port: Int, services: ServerServiceDefinition*): Server < (IO & Resources) =
    Resources.ensure {
      IO {
        val server = ServerBuilder.forPort(port)
          .addServices(services.asJava)
          .build()
          .start()
        (server, IO(server.shutdown()))
      }
    }

  /**
   * Executes a unary gRPC call with proper error handling.
   */
  def unaryCall[Req, Res](
      call: (Req, Metadata) => ListenableFuture[Res],
      request: Req,
      metadata: Metadata = new Metadata()
  ): Res < (IO & Aborts[StatusException]) =
    IO {
      val future = call(request, metadata)
      val result = future.get()
      result
    }.catching[StatusException]

  /**
   * Executes a server-streaming gRPC call.
   */
  def serverStreaming[Req, Res](
      call: (Req, Metadata) => Iterator[Res],
      request: Req,
      metadata: Metadata = new Metadata()
  ): Stream[Res, IO & Aborts[StatusException]] =
    Stream(IO {
      val iterator = call(request, metadata)
      new Iterator[Res]:
        def hasNext = iterator.hasNext
        def next() = iterator.next()
    }.catching[StatusException])

  /**
   * Executes a client-streaming gRPC call.
   */
  def clientStreaming[Req, Res](
      call: StreamObserver[Res] => StreamObserver[Req],
      requests: Stream[Req, Any],
      metadata: Metadata = new Metadata()
  ): Res < (IO & Aborts[StatusException]) =
    IO {
      val responseObserver = new StreamObserver[Res]:
        val result = new AtomicReference[Option[Res]](None)
        val error = new AtomicReference[Option[StatusException]](None)
        val latch = new java.util.concurrent.CountDownLatch(1)

        def onNext(value: Res): Unit = result.set(Some(value))
        def onError(t: Throwable): Unit =
          error.set(Some(StatusException.fromThrowable(t)))
          latch.countDown()
        def onCompleted(): Unit = latch.countDown()

      val requestObserver = call(responseObserver)
      requests.run.foreach { req =>
        requestObserver.onNext(req)
      }
      requestObserver.onCompleted()
      latch.await()

      error.get() match
        case Some(e) => throw e
        case None    => result.get().get
    }.catching[StatusException]

  /**
   * Executes a bidirectional streaming gRPC call.
   */
  def bidiStreaming[Req, Res](
      call: StreamObserver[Res] => StreamObserver[Req],
      requests: Stream[Req, Any],
      metadata: Metadata = new Metadata()
  ): Stream[Res, IO & Aborts[StatusException]] =
    Stream(IO {
      val output = new java.util.concurrent.BlockingQueue[Either[StatusException, Option[Res]]]
      val responseObserver = new StreamObserver[Res]:
        def onNext(value: Res): Unit = output.put(Right(Some(value)))
        def onError(t: Throwable): Unit = output.put(Left(StatusException.fromThrowable(t)))
        def onCompleted(): Unit = output.put(Right(None))

      val requestObserver = call(responseObserver)
      requests.run.foreach { req =>
        requestObserver.onNext(req)
      }
      requestObserver.onCompleted()

      new Iterator[Res]:
        def hasNext: Boolean =
          output.peek() match
            case null => false
            case Left(_) => false
            case Right(Some(_)) => true
            case Right(None) => false

        def next(): Res =
          output.take() match
            case Left(e) => throw e
            case Right(Some(v)) => v
            case Right(None) => throw new NoSuchElementException("No more elements")
    }.catching[StatusException])

  /**
   * Type class for creating gRPC stubs.
   */
  trait GrpcStubFactory[T]:
    def create(channel: ManagedChannel): T

  object GrpcStubFactory:
    given forBlockingStub[T <: AbstractBlockingStub[T]]: GrpcStubFactory[T] =
      (channel: ManagedChannel) => ???

    given forFutureStub[T <: AbstractFutureStub[T]]: GrpcStubFactory[T] =
      (channel: ManagedChannel) => ???

  /**
   * Wraps a StatusException for Kyo error handling.
   */
  case class StatusException(status: Status, metadata: Metadata, cause: Option[Throwable] = None)
      extends Exception(status.getDescription, cause.orNull)

  object StatusException:
    def fromThrowable(t: Throwable): StatusException = t match
      case se: io.grpc.StatusException =>
        StatusException(se.getStatus, se.getTrailers)
      case se: io.grpc.StatusRuntimeException =>
        StatusException(se.getStatus, se.getTrailers)
      case other =>
        StatusException(Status.INTERNAL.withDescription(other.getMessage), new Metadata(), Some(other))

end Grpc

/**
 * Internal utilities for gRPC support.
 */
package kyo.grpc.internal:

  import kyo.*

  private[grpc] object Resources:
    def ensure[T](f: => (T, IO[Unit]) < IO): T < (IO & kyo.Resources) =
      IO {
        val (resource, release) = f.pure
        kyo.Resources.addFinalizer(release)
        resource
      }

end kyo.grpc.internal
