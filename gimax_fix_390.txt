package io.kyo.grpc

import io.grpc.{Server, ServerBuilder, ServerServiceDefinition}
import io.grpc.stub.StreamObserver
import io.kyo.*
import io.kyo.direct.*
import scala.concurrent.{ExecutionContext, Future}
import scala.util.Try

trait KyoGrpcService[Req, Res]:
  def unary(request: Req): Req < (IO & Async)
  def serverStreaming(request: Req): Stream[Res, IO & Async]
  def clientStreaming(requests: Stream[Req, IO & Async]): Res < (IO & Async)
  def bidiStreaming(requests: Stream[Req, IO & Async]): Stream[Res, IO & Async]

object KyoGrpcServer:
  def apply(port: Int, services: List[ServerServiceDefinition])(using 
    ExecutionContext
  ): Server < (IO & Async) =
    IO {
      ServerBuilder
        .forPort(port)
        .addServices(services.asJava)
        .build()
        .start()
    }

  def start(server: Server): Unit < (IO & Async) =
    IO(server.start())

  def awaitTermination(server: Server): Unit < (IO & Async) =
    IO(server.awaitTermination())

package io.kyo.grpc

import io.grpc.{Channel, ManagedChannelBuilder}
import io.grpc.stub.{AbstractStub, ClientCalls, StreamObserver}
import io.kyo.*
import io.kyo.direct.*
import scala.concurrent.{ExecutionContext, Future}
import scala.util.Try

class KyoGrpcClient[Stub <: AbstractStub[Stub]](
  stub: Stub,
  channel: ManagedChannel
)(using ExecutionContext):

  def unaryCall[Req, Res](
    request: Req,
    call: (Stub, Req) => Future[Res]
  ): Res < (IO & Async) =
    IO.fromFuture(_ => call(stub, request))

  def serverStreamingCall[Req, Res](
    request: Req,
    call: (Stub, Req) => Iterator[Res]
  ): Stream[Res, IO & Async] =
    Stream.fromIterator(IO(call(stub, request)))

  def clientStreamingCall[Req, Res](
    requests: Stream[Req, IO & Async],
    call: (Stub, Iterator[Req]) => Future[Res]
  ): Res < (IO & Async) =
    IO.fromFuture(_ => call(stub, requests.iterator))

  def bidiStreamingCall[Req, Res](
    requests: Stream[Req, IO & Async],
    call: (Stub, Iterator[Req]) => Iterator[Res]
  ): Stream[Res, IO & Async] =
    Stream.fromIterator(IO(call(stub, requests.iterator)))

  def shutdown(): Unit < (IO & Async) =
    IO(channel.shutdown())

package io.kyo.grpc

import helloworld.*
import io.grpc.stub.StreamObserver
import io.grpc.{ServerServiceDefinition, Status}
import io.kyo.*
import io.kyo.direct.*
import scala.concurrent.{ExecutionContext, Future}

class GreeterService(using ExecutionContext) extends GreeterGrpc.Greeter:

  override def sayHello(
    request: HelloRequest,
    responseObserver: StreamObserver[HelloReply]
  ): Unit =
    val effect: HelloReply < (IO & Async) =
      for
        _ <- IO.unit
        reply = HelloReply(s"Hello, ${request.name}!")
      yield reply

    effect.onComplete {
      case Success(reply) =>
        responseObserver.onNext(reply)
        responseObserver.onCompleted()
      case Failure(ex) =>
        responseObserver.onError(ex)
    }

  override def sayHelloStream(
    responseObserver: StreamObserver[HelloReply]
  ): StreamObserver[HelloRequest] =
    new StreamObserver[HelloRequest]:
      override def onNext(value: HelloRequest): Unit =
        val reply = HelloReply(s"Hello, ${value.name}!")
        responseObserver.onNext(reply)

      override def onError(t: Throwable): Unit =
        responseObserver.onError(t)

      override def onCompleted(): Unit =
        responseObserver.onCompleted()

object GreeterService:
  def apply()(using ExecutionContext): ServerServiceDefinition =
    GreeterGrpc.bindService(new GreeterService(), _)
