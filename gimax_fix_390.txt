package helloworld;

// src/main/scala/io/kyo/grpc/GrpcServer.scala
package io.kyo.grpc

import io.grpc.{ServerBuilder, ServerServiceDefinition}
import io.kyo.*
import io.kyo.io.*
import scala.concurrent.{ExecutionContext, Future}

trait GrpcServer:
  def start(port: Int, services: ServerServiceDefinition*)(using IO): Unit < IO

object GrpcServer:
  val live: GrpcServer = new GrpcServer:
    def start(port: Int, services: ServerServiceDefinition*)(using IO): Unit < IO =
      IO:
        val server = ServerBuilder
          .forPort(port)
          .addServices(services.asJava)
          .build()
          .start()
        println(s"gRPC server started on port $port")
        server.awaitTermination()

    def channel(host: String, port: Int)(using IO): Channel < IO =
      IO:
        ManagedChannelBuilder
          .forAddress(host, port)
          .usePlaintext()
          .build()

// src/main/scala/io/kyo/grpc/GrpcKyo.scala
package io.kyo.grpc

import io.grpc.stub.AbstractStub
import io.kyo.*
import io.kyo.io.*
import scala.concurrent.{ExecutionContext, Future}
import scala.util.Try

object GrpcKyo:
  def fromFuture[A](future: => Future[A])(using IO): A < IO =
    IO:
      scala.concurrent.Await.result(future, scala.concurrent.duration.Duration.Inf)

  def stub[S <: AbstractStub[S]](stub: S): S =
    stub

// src/main/scala/io/kyo/grpc/GreeterService.scala
package io.kyo.grpc

import io.grpc.stub.StreamObserver
import io.kyo.*
import io.kyo.io.*
import helloworld.*

class GreeterService(using IO) extends GreeterGrpc.Greeter:
  override def sayHello(
      request: HelloRequest,
      responseObserver: StreamObserver[HelloReply]
  ): Unit =
    val effect: HelloReply < IO = IO:
      val name = request.name
      val message = s"Hello, $name!"
      HelloReply(message)

    IO.run(effect).onComplete:
      case scala.util.Success(reply) =>
        responseObserver.onNext(reply)
        responseObserver.onCompleted()
      case scala.util.Failure(exception) =>
        responseObserver.onError(exception)

// src/test/scala/io/kyo/grpc/GrpcSpec.scala
package io.kyo.grpc

import io.grpc.{ServerBuilder, ManagedChannelBuilder}
import io.kyo.*
import io.kyo.io.*
import helloworld.*
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import scala.concurrent.{ExecutionContext, Future}

class GrpcSpec extends AnyFlatSpec with Matchers:
  "gRPC server and client" should "work with Kyo" in {
    val test = IO:
      val server = ServerBuilder
        .forPort(0)
        .addService(new GreeterService())
        .build()
        .start()

      val port = server.getPort()
      val channel = ManagedChannelBuilder
        .forAddress("localhost", port)
        .usePlaintext()
        .build()

      val stub = GreeterGrpc.stub(channel)
      val request = HelloRequest("Kyo")
      val response = GrpcKyo.fromFuture(stub.sayHello(request))

      response.message should be("Hello, Kyo!")

      channel.shutdown()
      server.shutdown()

    IO.run(test)
  }

    val test = IO:
      val serverEffect = GrpcServer.live.start(0, new GreeterService())
