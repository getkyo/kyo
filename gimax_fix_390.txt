package kyogrpc

import io.grpc.{Server, ServerBuilder}
import io.grpc.stub.StreamObserver
import kyo.*
import kyo.direct.*
import scala.concurrent.{ExecutionContext, Future}
import scalapb.grpc.Grpc

class KyoGrpcServer(port: Int, greeterService: GreeterService):
  private var server: Server = null

  def start(): Unit < IOs =
    IOs {
      server = ServerBuilder
        .forPort(port)
        .addService(GreeterGrpc.bindService(greeterService, ExecutionContext.global))
        .build()
        .start()
      println(s"Server started on port $port")
      server
    }

  def blockUntilShutdown(): Unit < IOs =
    IOs {
      if server != null then
        server.awaitTermination()
    }

  def shutdown(): Unit < IOs =
    IOs {
      if server != null then
        server.shutdown()
    }

// src/main/scala/kyogrpc/GreeterService.scala
package kyogrpc

import io.grpc.stub.StreamObserver
import kyo.*
import kyo.direct.*
import scala.concurrent.{ExecutionContext, Future}
import scalapb.grpc.Grpc

class GreeterService extends GreeterGrpc.Greeter:
  def sayHello(request: HelloRequest): Future[HelloReply] =
    Grpc.guavaFutureToScalaFuture(
      kyoToGuavaFuture(handleSayHello(request))
    )

  private def handleSayHello(request: HelloRequest): HelloReply < IOs =
    IOs {
      val reply = HelloReply(message = s"Hello, ${request.name}!")
      reply
    }

  private def kyoToGuavaFuture[A](effect: A < IOs): com.google.common.util.concurrent.ListenableFuture[A] =
    val promise = com.google.common.util.concurrent.SettableFuture.create[A]()
    IOs.run {
      effect.map { result =>
        promise.set(result)
        result
      }.onError { ex =>
        promise.setException(ex)
      }
    }
    promise

package kyogrpc

import io.grpc.{ManagedChannel, ManagedChannelBuilder}
import kyo.*
import kyo.direct.*

class KyoGrpcClient(host: String, port: Int):
  private var channel: ManagedChannel = null
  private var stub: GreeterGrpc.GreeterBlockingStub = null

  def start(): Unit < IOs =
    IOs {
      channel = ManagedChannelBuilder.forAddress(host, port).usePlaintext().build()
      stub = GreeterGrpc.blockingStub(channel)
    }

  def sayHello(name: String): String < IOs =
    IOs {
      val request = HelloRequest(name = name)
      val reply = stub.sayHello(request)
      reply.message
    }

  def shutdown(): Unit < IOs =
    IOs {
      if channel != null then
        channel.shutdown()
    }

package kyogrpc

import kyo.*
import kyo.direct.*
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import scala.concurrent.{Await, Future}
import scala.concurrent.duration.*

class KyoGrpcSpec extends AnyFlatSpec with Matchers:
  "Kyo gRPC" should "handle request-response correctly" in {
    val server = new KyoGrpcServer(50051, new GreeterService())
    val client = new KyoGrpcClient("localhost", 50051)

    val testEffect = for
      _ <- server.start()
      _ <- client.start()
      response <- client.sayHello("World")
      _ <- client.shutdown()
      _ <- server.shutdown()
    yield response

    val result = IOs.run(testEffect)
    result shouldBe "Hello, World!"
  }

  it should "handle multiple requests" in {
    val server = new KyoGrpcServer(50052, new GreeterService())
    val client = new KyoGrpcClient("localhost", 50052)

    val testEffect = for
      _ <- server.start()
      _ <- client.start()
      r1 <- client.sayHello("Alice")
      r2 <- client.sayHello("Bob")
      _ <- client.shutdown()
      _ <- server.shutdown()
    yield (r1, r2)

    val (r1, r2) = IOs.run(testEffect)
    r1 shouldBe "Hello, Alice!"
    r2 shouldBe "Hello, Bob!"
  }

  it should "handle errors gracefully" in {
    val server = new KyoGrpcServer(50053, new GreeterService())
    val client = new KyoGrpcClient("localhost", 50053)

    val testEffect = for
      _ <- server.start()
      _ <- client.start()
      _ <- server.shutdown() // Shutdown server before client request
      result <- client.sayHello("World").attempt
      _ <- client.shutdown()
    yield result

    val result = IOs.run(testEffect)
    result.isLeft shouldBe true
  }
