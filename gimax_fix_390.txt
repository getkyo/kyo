/**
 * Kyo gRPC support module providing integration with ScalaPB generated code.
 * 
 * This module enables efficient gRPC client and server implementations using Kyo's
 * effect system, with performance optimizations over standard ScalaPB gRPC.
 */
object Grpc {

  /**
   * Creates a Kyo-based gRPC client stub for a given service.
   * 
   * @tparam T The service trait type (generated by ScalaPB)
   * @param channel The gRPC channel
   * @param options Call options for the stub
   * @return A Kyo effect that produces the client stub
   */
  def client[T](channel: Channel, options: CallOptions = CallOptions.DEFAULT)(using
      descriptor: ServiceDescriptor[T],
      tag: Tag[T]
  ): T < (IO & Async) =
    IO {
      descriptor.client(channel, options)
    }

  /**
   * Creates a Kyo-based gRPC server that can handle ScalaPB services.
   * 
   * @param services The services to register
   * @param port The server port
   * @return A Kyo effect that manages the server lifecycle
   */
  def server(services: Seq[BindableService], port: Int = 8080): Server < (IO & Async) =
    IO {
      val serverBuilder = ServerBuilder.forPort(port)
      services.foreach(serverBuilder.addService)
      val server = serverBuilder.build()
      server.start()
      server
    }

  /**
   * Unary call support for Kyo gRPC.
   */
  object unary {
    def apply[Req, Res](stub: Req => Future[Res], request: Req)(using
        Frame
    ): Res < (IO & Async) =
      IO {
        val p = Promise[Res]()
        stub(request).onComplete {
          case Success(value) => p.success(value)
          case Failure(ex)   => p.fail(ex)
        }(using ExecutionContext.parasitic)
        p.get
      }.flatMap(identity)
  }

  /**
   * Server streaming call support for Kyo gRPC.
   */
  object serverStreaming {
    def apply[Req, Res](stub: Req => Iterator[Res], request: Req)(using
        Frame
    ): Stream[Res, IO & Async] < (IO & Async) =
      IO {
        val iterator = stub(request)
        Stream(iterator)
      }
  }

  /**
   * Client streaming call support for Kyo gRPC.
   */
  object clientStreaming {
    def apply[Req, Res](stub: StreamObserver[Req] => Future[Res], stream: Stream[Req, Any])(using
        Frame
    ): Res < (IO & Async) =
      IO {
        val p = Promise[Res]()
        val observer = new StreamObserver[Req] {
          def onNext(value: Req): Unit = ()
          def onError(t: Throwable): Unit = p.fail(t)
          def onCompleted(): Unit = ()
        }
        val future = stub(observer)
        stream.runForeach { req =>
          IO(observer.onNext(req))
        }.andThen {
          IO(observer.onCompleted())
        }.andThen {
          future.onComplete {
            case Success(value) => p.success(value)
            case Failure(ex)   => p.fail(ex)
          }(using ExecutionContext.parasitic)
          p.get
        }
      }.flatMap(identity)
  }

  /**
   * Bidi streaming call support for Kyo gRPC.
   */
  object bidiStreaming {
    def apply[Req, Res](
        stub: StreamObserver[Res] => StreamObserver[Req],
        stream: Stream[Req, Any]
    )(using Frame): Stream[Res, IO & Async] < (IO & Async) =
      IO {
        val responses = new AtomicReference[List[Res]](Nil)
        val p = Promise[Unit]()
        val responseObserver = new StreamObserver[Res] {
          def onNext(value: Res): Unit = {
            val current = responses.get()
            while (!responses.compareAndSet(current, value :: current)) {}
          }
          def onError(t: Throwable): Unit = p.fail(t)
          def onCompleted(): Unit = p.success(())
        }
        val requestObserver = stub(responseObserver)
        stream.runForeach { req =>
          IO(requestObserver.onNext(req))
        }.andThen {
          IO(requestObserver.onCompleted())
        }.andThen {
          p.get.map(_ => Stream.fromIterable(responses.get().reverse))
        }
      }.flatMap(identity)
  }

  /**
   * Service descriptor for ScalaPB generated services.
   */
  trait ServiceDescriptor[T] {
    def client(channel: Channel, options: CallOptions): T
    def bindService(impl: T): ServerServiceDefinition
  }

  object ServiceDescriptor {
    inline given derived[T](using m: scala.deriving.Mirror.SumOf[T]): ServiceDescriptor[T] =
      new ServiceDescriptor[T] {
        def client(channel: Channel, options: CallOptions): T =
          throw new UnsupportedOperationException("Derivation not supported for sum types")
        def bindService(impl: T): ServerServiceDefinition =
          throw new UnsupportedOperationException("Derivation not supported for sum types")
      }
  }

  /**
   * Internal utilities for Kyo gRPC.
   */
  private[grpc] object internal {
    def toKyo[A](future: Future[A])(using Frame): A < (IO & Async) =
      IO {
        val p = Promise[A]()
        future.onComplete {
          case Success(value) => p.success(value)
          case Failure(ex)   => p.fail(ex)
        }(using ExecutionContext.parasitic)
        p.get
      }.flatMap(identity)

    def fromKyo[A](effect: A < (IO & Async))(using Frame): Future[A] =
      IO {
        val p = scala.concurrent.Promise[A]()
        kyo.Async.run(effect).map { fiber =>
          fiber.onComplete {
            case Success(value) => p.success(value)
            case Failure(ex)   => p.failure(ex)
          }(using ExecutionContext.parasitic)
          p.future
        }
      }.pipe(kyo.IO.run).eval
  }
}
