/**
 * Kyo gRPC support module providing integration with ScalaPB generated code.
 * 
 * This module enables gRPC client and server implementations using Kyo's effect system,
 * providing efficient, composable gRPC services with proper resource handling and backpressure.
 */
object Grpc:

  /**
   * Configuration for gRPC channel and server settings.
   */
  case class Config(
    deadline: Option[Duration] = None,
    maxRetries: Int = 3,
    retryDelay: Duration = 100.millis,
    compression: Option[String] = None,
    interceptors: List[ClientInterceptor] = Nil
  )

  object Config:
    val default: Config = Config()

  /**
   * Kyo effect type for gRPC operations.
   */
  type GrpcIO[T] = Async > (Abort[StatusException] | Resource)

  /**
   * Creates a gRPC client stub for the given service.
   * 
   * @tparam T The ScalaPB service stub type
   * @param channel The gRPC ManagedChannel
   * @param config Client configuration
   * @return A Kyo effect that produces the client stub
   */
  def client[T](channel: ManagedChannel, config: Config = Config.default)(
    using stubFactory: GrpcStubFactory[T]
  ): T < GrpcIO =
    Resource.ensure(IO(channel.shutdown())) *>
      IO {
        val stub = stubFactory.create(channel)
        val configured = config.interceptors.foldLeft(stub) { (s, interceptor) =>
          stubFactory.intercept(s, interceptor)
        }
        configured
      }

  /**
   * Creates a gRPC server that serves the given service implementation.
   * 
   * @tparam T The ScalaPB service type
   * @param service The service implementation
   * @param port The server port
   * @param config Server configuration
   * @return A Kyo effect that manages the server lifecycle
   */
  def server[T](
    service: T,
    port: Int = 0,
    config: ServerConfig = ServerConfig.default
  )(
    using serviceFactory: GrpcServiceFactory[T]
  ): Server < GrpcIO =
    IO {
      val serverBuilder = ServerBuilder.forPort(port)
        .addService(serviceFactory.create(service))
        .executor(config.executor.getOrElse(kyo.concurrent.ThreadPool.default))
      
      config.interceptors.foreach(serverBuilder.intercept)
      
      val server = serverBuilder.build().start()
      server
    }.flatMap { server =>
      Resource.ensure(IO(server.shutdown()).map(_ => server.awaitTermination()))
        .andThen(server)
    }

  /**
   * Runs a gRPC call with proper error handling and retries.
   */
  def call[A, B](request: A, stub: B, config: Config = Config.default)(
    using callable: GrpcCallable[A, B]
  ): A < GrpcIO =
    def attempt(retriesLeft: Int): A < GrpcIO =
      IO {
        callable.call(request, stub)
      }.map { result =>
        result
      }.catch {
        case ex: StatusException if retriesLeft > 0 && isRetryable(ex.getStatus) =>
          IO.sleep(config.retryDelay) *> attempt(retriesLeft - 1)
        case ex: StatusException =>
          Abort.fail(ex)
      }

    attempt(config.maxRetries)

  private def isRetryable(status: Status): Boolean =
    status.getCode match {
      case Status.Code.UNAVAILABLE | Status.Code.DEADLINE_EXCEEDED |
           Status.Code.RESOURCE_EXHAUSTED | Status.Code.ABORTED =>
        true
      case _ => false
    }

  /**
   * Server configuration.
   */
  case class ServerConfig(
    executor: Option[kyo.concurrent.ThreadPool] = None,
    interceptors: List[ServerInterceptor] = Nil,
    maxInboundMessageSize: Option[Int] = None
  )

  object ServerConfig:
    val default: ServerConfig = ServerConfig()

  /**
   * Type class for creating gRPC stubs.
   */
  trait GrpcStubFactory[T]:
    def create(channel: ManagedChannel): T
    def intercept(stub: T, interceptor: ClientInterceptor): T

  /**
   * Type class for creating gRPC service implementations.
   */
  trait GrpcServiceFactory[T]:
    def create(service: T): BindableService

  /**
   * Type class for making gRPC calls.
   */
  trait GrpcCallable[A, B]:
    def call(request: A, stub: B): A < IO

end Grpc

/**
 * Internal utilities for gRPC support.
 */
package kyo.grpc.internal:

  import kyo.*
  import io.grpc.*

  private[grpc] object GrpcInternal:

    /**
     * Converts a gRPC StatusException to a Kyo Abort.
     */
    def toAbort[T](status: Status, description: String = ""): T < Abort[StatusException] =
      Abort.fail(status.withDescription(description).asRuntimeException())

    /**
     * Wraps a gRPC future-style call into a Kyo effect.
     */
    def futureToKyo[T](future: com.google.common.util.concurrent.ListenableFuture[T]): T < Async =
      Async.fromFuture(
        scala.concurrent.Future {
          future.get()
        }(scala.concurrent.ExecutionContext.global)
      )

    /**
     * Manages gRPC call cancellation via Kyo resources.
     */
    def cancellableCall[T](call: ClientCall[_, _], effect: T < Async): T < (Async | Resource) =
      Resource.ensure(IO(call.cancel("Kyo cancellation", null))).andThen(effect)

  end GrpcInternal

end kyo.grpc.internal
