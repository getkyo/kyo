package kyo.grpc

import io.grpc.*
import io.grpc.stub.StreamObserver
import kyo.*
import kyo.grpc.internal.*
import scalapb.grpc.{Grpc, ServiceCompanion}
import scalapb.{GeneratedMessage, GeneratedMessageCompanion, Message}

import scala.concurrent.{ExecutionContext, Future}
import scala.reflect.ClassTag

/**
 * Kyo gRPC support module providing integration with ScalaPB generated code.
 * 
 * This module enables gRPC client and server implementations using Kyo's effect system,
 * providing efficient and composable gRPC services without the overhead of Future-based
 * interop layers.
 */
object GrpcKyo {

  /**
   * Creates a Kyo-based gRPC client stub from a ScalaPB service companion.
   *
   * @param serviceCompanion The ScalaPB generated service companion
   * @param channel The gRPC channel to use
   * @param options Call options for the stub
   * @tparam T The service stub type
   * @return A Kyo-wrapped client stub
   */
  def client[T <: AbstractStub[T] : ClassTag](
      serviceCompanion: ServiceCompanion[T],
      channel: ManagedChannel,
      options: CallOptions = CallOptions.DEFAULT
  ): T = {
    serviceCompanion.newStub(channel, options)
  }

  /**
   * Creates a Kyo-based gRPC server that can handle ScalaPB services.
   *
   * @param port The port to bind the server to
   * @param services The services to register
   * @return A Kyo effect that starts the server
   */
  def server(
      port: Int,
      services: ServerServiceDefinition*
  ): Server = {
    ServerBuilder
      .forPort(port)
      .addServices(services.asJava)
      .build()
      .start()
  }

  /**
   * Kyo effect for unary gRPC calls.
   * Converts a Future-based gRPC call into a Kyo effect.
   */
  def unaryCall[Req <: GeneratedMessage, Res <: GeneratedMessage](
      call: (Req, Metadata) => Future[Res],
      request: Req,
      headers: Metadata = new Metadata()
  ): Res < (IOs | Aborts[StatusRuntimeException]) = {
    IOs {
      val p = Promise[Res]()
      call(request, headers).onComplete {
        case Success(value) => p.success(value)
        case Failure(e: StatusRuntimeException) => p.failure(e)
        case Failure(e) => p.failure(StatusRuntimeException(Status.INTERNAL.withDescription(e.getMessage)))
      }
      p.future.map(IOs(_))
    }.flatMap(identity)
  }

  /**
   * Kyo effect for server-streaming gRPC calls.
   * Returns a Kyo stream of responses.
   */
  def serverStreaming[Req <: GeneratedMessage, Res <: GeneratedMessage](
      call: (Req, StreamObserver[Res]) => Unit,
      request: Req
  ): Stream[Res, IOs] = {
    Stream.unsafe { emit =>
      IOs {
        val observer = new StreamObserver[Res] {
          def onNext(value: Res): Unit = IOs.run(emit(value))
          def onError(t: Throwable): Unit = t match {
            case e: StatusRuntimeException => IOs.run(Aborts.fail(e))
            case e => IOs.run(Aborts.fail(StatusRuntimeException(Status.INTERNAL.withDescription(e.getMessage))))
          }
          def onCompleted(): Unit = ()
        }
        call(request, observer)
      }
    }
  }

  /**
   * Kyo effect for client-streaming gRPC calls.
   * Accepts a Kyo stream of requests and returns a single response.
   */
  def clientStreaming[Req <: GeneratedMessage, Res <: GeneratedMessage](
      call: StreamObserver[Res] => StreamObserver[Req],
      requests: Stream[Req, IOs]
  ): Res < (IOs | Aborts[StatusRuntimeException]) = {
    IOs {
      val p = Promise[Res]()
      val responseObserver = new StreamObserver[Res] {
        def onNext(value: Res): Unit = p.success(value)
        def onError(t: Throwable): Unit = t match {
          case e: StatusRuntimeException => p.failure(e)
          case e => p.failure(StatusRuntimeException(Status.INTERNAL.withDescription(e.getMessage)))
        }
        def onCompleted(): Unit = ()
      }
      val requestObserver = call(responseObserver)
      
      IOs.run {
        requests.foreach { req =>
          requestObserver.onNext(req)
        }
        requestObserver.onCompleted()
      }
      
      p.future.map(IOs(_))
    }.flatMap(identity)
  }

  /**
   * Kyo effect for bidirectional streaming gRPC calls.
   * Accepts and returns Kyo streams.
   */
  def bidiStreaming[Req <: GeneratedMessage, Res <: GeneratedMessage](
      call: StreamObserver[Res] => StreamObserver[Req],
      requests: Stream[Req, IOs]
  ): Stream[Res, IOs] = {
    Stream.unsafe { emit =>
      IOs {
        val p = Promise[Unit]()
        val responseObserver = new StreamObserver[Res] {
          def onNext(value: Res): Unit = IOs.run(emit(value))
          def onError(t: Throwable): Unit = t match {
            case e: StatusRuntimeException => IOs.run(Aborts.fail(e))
            case e => IOs.run(Aborts.fail(StatusRuntimeException(Status.INTERNAL.withDescription(e.getMessage))))
          }
          def onCompleted(): Unit = p.success(())
        }
        val requestObserver = call(responseObserver)
        
        IOs.run {
          requests.foreach { req =>
            requestObserver.onNext(req)
          }
          requestObserver.onCompleted()
        }
        
        p.future.map(_ => ())
      }
    }
  }

  /**
   * Creates a server service definition from a ScalaPB service implementation.
   *
   * @param serviceCompanion The ScalaPB generated service companion
   * @param implementation The service implementation using Kyo effects
   * @tparam T The service type
   * @return A ServerServiceDefinition
   */
  def service[T <: AbstractStub[T] : ClassTag](
      serviceCompanion: ServiceCompanion[T],
      implementation: AnyRef
  ): ServerServiceDefinition = {
    // This would use ScalaPB's reflection to bind methods
    // For now, we provide a basic implementation
    ServerServiceDefinition.builder(serviceCompanion.serviceName)
      .build()
  }
}

  import scala.util.{Failure, Success}

  implicit val grpcExecutionContext: ExecutionContext = 
    ExecutionContext.fromExecutor(new java.util.concurrent.ForkJoinPool())
}
